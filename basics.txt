//basic of c-language
//array
#include<stdio.h>//header file
#include<stdlib.h>
int main()
{
    int b[5]={1,2,3,4,5};//creating array in stack
    int *a;//creating array in heap 
    int n,i;
    printf("enter n value\n");//size of array
    scanf("%d",&n);
    for(i=0;i<n;i++)
    {
        printf("%d ",b[i]);//result of b
    }
    printf("\n");
    a=(int *)malloc(sizeof(int));
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);//taking elements from user
    }
    for(i=0;i<n;i++)
    {
        printf("%d ",a[i]);//result of a
    }
    return 0;
}
